// FILE: programs/dloom_stake/src/instructions/compound.rs

use anchor_lang::prelude::*;
use anchor_spl::token::{self, Token, Transfer};
use crate::{
    errors::StakingError,
    events::Compounded,
    state::{Farm, Staker},
    instructions::{
        claim_rewards::ClaimRewards,
        reward_math::{update_reward_accumulator, calculate_pending_rewards},
    },
};

pub fn handle_compound(ctx: Context<ClaimCompound>) -> Result<()> {
    let farm = &mut ctx.accounts.update_stake.farm;
    let staker = &mut ctx.accounts.update_stake.staker;

    require!(
        farm.reward_mint == farm.lp_mint,
        StakingError::CompoundingNotSupported
    );
    
    update_reward_accumulator(farm)?;
    let pending_rewards = calculate_pending_rewards(farm, staker)?;
    require!(pending_rewards > 0, StakingError::NoRewardsToClaim);

    staker.rewards_paid = 0;
    staker.reward_per_token_snapshot = farm.reward_per_token_stored;

    // --- LOGIC DIFFERENCE FROM CLAIM ---
    // Instead of transferring rewards to the user, we transfer them from the
    // reward vault directly to the LP vault, effectively "staking" them.

    // 4. Transfer rewards from the reward vault to the LP vault.
    let farm_key = farm.key();
    let seeds = &[
        b"farm".as_ref(),
        farm.lp_mint.as_ref(),
        farm.reward_mint.as_ref(),
        &[farm.bump],
    ];
    let signer_seeds = &[&seeds[..]];

    let cpi_accounts = Transfer {
        from: ctx.accounts.reward_vault.to_account_info(),
        to: ctx.accounts.update_stake.lp_vault.to_account_info(),
        authority: farm.to_account_info(),
    };
    let cpi_program = ctx.accounts.update_stake.token_program.to_account_info();
    let cpi_ctx = CpiContext::new_with_signer(cpi_program, cpi_accounts, signer_seeds);

    token::transfer(cpi_ctx, pending_rewards)?;

    // 5. Update farm and staker state with the new staked balance.
    // First, remove old weighted stake.
    let old_weighted_stake = (staker.balance as u128)
        .checked_mul(staker.reward_multiplier as u128)
        .ok_or(StakingError::MathOverflow)?
        .checked_div(10000)
        .ok_or(StakingError::MathOverflow)?;
    farm.total_weighted_stake = farm.total_weighted_stake
        .checked_sub(old_weighted_stake)
        .ok_or(StakingError::MathOverflow)?;
        
    // Add compounded amount to staker's balance.
    staker.balance = staker.balance.checked_add(pending_rewards).ok_or(StakingError::MathOverflow)?;
    
    // Add the new total weighted stake back to the farm's total.
    let new_weighted_stake = (staker.balance as u128)
        .checked_mul(staker.reward_multiplier as u128)
        .ok_or(StakingError::MathOverflow)?
        .checked_div(10000)
        .ok_or(StakingError::MathOverflow)?;
    farm.total_weighted_stake = farm.total_weighted_stake
        .checked_add(new_weighted_stake)
        .ok_or(StakingError::MathOverflow)?;

    // 6. Emit the event.
    emit!(Compounded {
        owner: staker.owner,
        farm_address: farm.key(),
        reward_amount_compounded: pending_rewards,
        new_total_staked_balance: staker.balance,
    });

    Ok(())
}
