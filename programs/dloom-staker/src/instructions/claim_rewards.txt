use anchor_lang::prelude::*;
use anchor_spl::{
    associated_token::AssociatedToken,
    token::{self, Mint, Token, TokenAccount, Transfer},
};
use crate::{
    errors::StakingError,
    events::RewardsClaimed,
    state::{Farm, Staker},
    instructions::{
        stake::UpdateStake,
        reward_math::{update_reward_accumulator, calculate_pending_rewards},
    },
};

pub fn handle_claim_rewards(ctx: Context<ClaimRewards>) -> Result<()> {
    let farm = &mut ctx.accounts.farm;
    let staker = &mut ctx.accounts.staker;

    update_reward_accumulator(farm)?;

    let pending_rewards = calculate_pending_rewards(farm, staker)?;
    require!(pending_rewards > 0, StakingError::NoRewardsToClaim);

    staker.rewards_paid = 0;
    staker.reward_per_token_snapshot = farm.reward_per_token_stored;

    let farm_key = farm.key();
    let seeds = &[
        b"farm".as_ref(),
        farm.lp_mint.as_ref(),
        farm.reward_mint.as_ref(),
        &[farm.bump],
    ];
    let signer_seeds = &[&seeds[..]];

    let cpi_accounts = Transfer {
        from: ctx.accounts.reward_vault.to_account_info(),
        to: ctx.accounts.user_reward_token_account.to_account_info(),
        authority: farm.to_account_info(),
    };
    let cpi_program = ctx.accounts.token_program.to_account_info();
    let cpi_ctx = CpiContext::new_with_signer(cpi_program, cpi_accounts, signer_seeds);

    token::transfer(cpi_ctx, pending_rewards)?;

    // 5. Emit the event.
    emit!(RewardsClaimed {
        owner: staker.owner,
        farm_address: farm.key(),
        amount: pending_rewards,
    });

    Ok(())
}

#[derive(Accounts)]
pub struct ClaimRewardsContext<'info> {
    // Inherit all accounts from UpdateStake
    pub update_stake: UpdateStake<'info>,

    #[account(
        mut,
        address = update_stake.farm.reward_vault
    )]
    pub reward_vault: Account<'info, TokenAccount>,
    
    #[account(
        init_if_needed,
        payer = update_stake.owner,
        associated_token::mint = update_stake.farm.reward_mint,
        associated_token::authority = update_stake.owner,
    )]
    pub user_reward_token_account: Account<'info, TokenAccount>,

    pub associated_token_program: Program<'info, AssociatedToken>,
    pub system_program: Program<'info, System>,
}
